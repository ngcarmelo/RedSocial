en express 4.16 o superior body-parser esta incluido en sus metodos
Alberto · Clase 8 · hace 9 meses
a partir de la versión 4.16 de express body-parser esta incluido dentro de sus métodos.

//con body-parser
 
//importar libreria
const bodyParser = require('body-parser');
 
//uso
app.use(bodyParser.urlencoded({
  extended: true
}));
 
app.use(bodyParser.json());

//con express 4.16 en adelante
 
//importar libreria
const express = require('express');
 
//uso
app.use(express.urlencoded({
  extended: true
}));
 
app.use(express.json());
